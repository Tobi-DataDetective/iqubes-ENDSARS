#working on endsars23
corpus23<- iconv(endsars23$text)
#remove retweets
corpus23<- gsub("(RT|via)((?:\\b\\w*@\\w+)+)"," ",corpus23)
#remove people
corpus23<- gsub("@\\w+"," ",corpus23)
corpus23<- gsub("\n.*"," ",corpus23)

######creating a corpus###############
corpus23<- Corpus(VectorSource(corpus23))
corpus23<- tm_map(corpus23,PlainTextDocument)
#inspect(corpus23[1:5])

################cleaning text#####################
corpus23<-tm_map(corpus23,tolower)
#inspect(corpus23[1:5])
corpus23<-tm_map(corpus23,removePunctuation)
corpus23<-tm_map(corpus23,removeNumbers)

cleanset23<- tm_map(corpus23, removeWords,stopwords('english'))
cleanset23<- tm_map(cleanset23,stemDocument)
#inspect(cleanset23[1:5])
removeURL23 <- function(x) gsub('http[[:alnum:]]*','',x)
cleanset23<- tm_map(cleanset23,content_transformer(removeURL23))
cleanset23<- tm_map(cleanset23,removeWords,c('rt','RT'))

inspect(cleanset23[1:5])

###########Term document matrix##########
tdm23<- TermDocumentMatrix(cleanset23)
#tdm23
#tdm23<- as.matrix(tdm23)

tdm23<- as.matrix(tdm23)

freq23<-rowSums(tdm23)

freq23<-subset(freq23,freq23>=500)
freq23

#########Bar plot########
barplot(freq23,
        las = 2,
        col = rainbow(20),ylab = 'tweet frequency',main='23RD MARCH MOST FREQUENT WORDS')
